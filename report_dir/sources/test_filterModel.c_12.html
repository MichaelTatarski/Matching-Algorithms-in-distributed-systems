Filter *testFilter;
 
void setUp(void)
{

}

{

{
    filter_addSubFilterINT32(testFilter, "testString", GREATER_THAN, 2020);
    filter_addSubFilterINT32(testFilter, "2ndtestName", EQUALS, 50);
}

{
    filter_addSubFilterDOUBLE(testFilter, "testString", GREATER_THAN, 2020.1);
